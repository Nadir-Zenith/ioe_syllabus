\section*{Course Objective:}
To provide concept and knowledge of economic studies that will be useful for the evaluation engineering projects and make decisions related to investments.

\mylist{
    \textbf{Introduction \hfill (3 hours)}
    \mylist{
        Origin of Engineering Economy;
        Principles of Engineering Economy;
        Role of Engineers in Decision Making;
        Cash Flow Diagram
    };
    \textbf{Interest and Time Value of Money \hfill (6 hours)}
    \mylist{
        Introduction to Time Value of Money;
        Simple Interest;
        Compound Interest
        \mylist{
            Nominal Interest Rate;
            Effective Interest Rate;
            Continuous Compounding
        };
        Economic Equivalence;
        Development of Interest Formulas
        \mylist{
            The Five Types of Cash Flows;
            Single Cash Flow Formulas;
            Uneven Payment Series;
            Equal Payment Series;
            Linear Gradient Series;
            Geometric Gradient Series
        }
    };
    \textbf{Basic Methodologies of Engineering Economic Analysis \hfill (8 hours)}
    \mylist{
        Determining Minimum Attractive (Acceptable) Rate of Return (MARR);
        Payback Period Method;
        Equivalent Worth Methods
        \mylist{
            Present Worth Method;
            Future Worth Method;
            Annual Worth Method
        };
        Rate of Return Methods
        \mylist{
            Internal Rate of Return Method;
            External/Modified Rate of Return Method
        };
        Public Sector Economic Analysis (Benefit Cost Ratio Method);
        Introduction to Life-cycle Costing;
        Introduction to Financial and Economic Analysis
    };
    \textbf{Comparative Analysis of Alternatives \hfill (6 hours)}
    \mylist{
        Comparing Mutually Exclusive Alternatives having same useful life by
        \mylist{
            Payback Period Method and Equivalent Worth Method;
            Rate of Return Methods and Benefit Cost Ratio Method
        };
        Comparing Mutually Exclusive Alternatives having different useful lives by
        \mylist{
            Repeatability Assumption;
            Co-terminated Assumption;
            Capitalized Worth Method
        };
        Comparing Mutually Exclusive, contingent and independent projects in combination
    };
    \textbf{Replacement Analysis \hfill (6 hours)}
    \mylist{
        Fundamentals of Replacement Analysis
        \mylist{
            Basic Concepts and Terminology;
            Approaches for comparing defender and challenger
        };
        Economic service life of challenger and defender;
        Replacement analysis when required service life is long
        \mylist{
            Required Assumptions and decision framework;
            Replacement analysis under the infinite planning horizon;
            Replacement analysis under the finite planning horizon
        }
    };
    \textbf{Risk Analysis \hfill (6 hours)}
    \mylist{
        Origin/Sources of Project Risks;
        Methods of describing project risks
        \mylist{
            Sensitivity analysis;
            Breakeven analysis;
            Scenario analysis
        };
        Probability concept of Economic analysis;
        Decision Tree and Sequential Investment Decisions
    };
    \textbf{Depreciation and Corporate Income Taxes \hfill (6 hours)}
    \mylist{
        Concept and Terminology of Depreciation;
        Basic Method of Depreciation
        \mylist{
            Straight Line method;
            Declining Balance method;
            Sinking fund method;
            Sum of the year digit method;
            Modified Accelerated Cost Recovery System (MACRS)
        };
        Introduction to Corporate Income Tax;
        After Tax Cash Flow Estimate;
        General Procedure for making after tax economic analysis
    };
    \textbf{Inflation and its Impact on Project cash flows \hfill (4 hours)}
    \mylist{
        Concept of Inflation;
        Measuring Inflation;
        Equivalence Calculation under Inflation;
        Impact of Inflation on Economic Evaluation
    }
}
