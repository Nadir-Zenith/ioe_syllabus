\section*{Course Objective:}
To continue the study of modern communication systems, their characteristics and design.


\mylist{
    \textbf{Telecommunication Networks: \hfill (4 hours)}
    \mylist{
        Evolution of telecommunications;
        Classification of switching system
    };
    \textbf{Transmission Media \hfill (4 hours)}
    \mylist{
        Transmission media characteristics;
        Transmission lines;
        Hybrid Transformer and circuits;
        Signal and noise measurement
    };
    \textbf{Signal Multiplexing \hfill (4 hours)}
    \mylist{
        Frequency division multiplex, Wavelength division multiplex;
        Space division multiplex;
        Time division multiplex{;} North American TDM system, The European E1
    };
    \textbf{Signaling System \hfill (4 hours)}
    \mylist{
        Classification of Signaling Systems: Channel Associated Signaling and Common Channel Signaling;
        ITU Common Channel Signalling System \# 7 (SS7)
    };
    \textbf{Telephone Traffic \hfill (9 hours)}
    \mylist{
        Network Traffic load and parameters;
        Loss System: Grade of service(GoS) and Blocking probability;
        Delay System: Queuing theory;
        Routing;
        Numbering Plans, Charging Plans
    };
    \textbf{Telecommunication Regulations  \hfill (2 hours)}
    \mylist{
        Purpose of ITU ( International Telecommunications Union);
        NTA ( Nepal Telecommunications Authority)
    };
    \textbf{Data Communication \hfill (10 hours)}
    \mylist{
        Switching Techniques in data communication;
        IP Switching;
        Soft Switching;
        Routing and Flow control;
        ISDN;
        DSL
    }
}
